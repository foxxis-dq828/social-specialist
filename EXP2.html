# Softmax function for numerical stability and temperature scaling
tau_softmax <- function(x, tau) {
  exps <- exp(log(x) * tau)
  exps / sum(exps)
}

# Build a joint hypothesis table comparing two agents' beliefs
different_agent <- function(jar1, jar2, s1, s2, k1, k2,tau2) {
  # jar1, jar2: named numeric vectors with a 'red' element; total balls in each jar = 10
  # s1, s2: "Type1" or "Type2"
  total_balls <- 10
  
  # Possible counts of blue balls (at least 1 blue and 1 yellow)
  jar1_blue_range <- seq(1, total_balls - jar1["red"] - 1)
  jar2_blue_range <- seq(1, total_balls - jar2["red"] - 1)
  
  # Generate all combinations of (jar1_blue, jar2_blue, observed color, chosen jar)
  hypotheses <- expand.grid(
    jar1_blue = jar1_blue_range,
    jar2_blue = jar2_blue_range,
    ob = c("blue", "yellow", "red"),
    chosen_jar = 1:2,
    KEEP.OUT.ATTRS = FALSE
  ) |>
    mutate(
      # Calculate the full distribution for jar1 and jar2.
      # The yellow count is computed so that blue + yellow is constant
      jar1 = Map(
        function(b) c("blue" = b, 
                      "yellow" = (max(jar1_blue_range) + min(jar1_blue_range)) - b,
                      "red" = unname(jar1["red"])),
        jar1_blue
      ),
      jar2 = Map(
        function(b) c("blue" = b, 
                      "yellow" = (max(jar2_blue_range) + min(jar2_blue_range)) - b,
                      "red" = unname(jar2["red"])),
        jar2_blue
      ),
      prior_ob = pmap_dbl(
        list(jar1, jar2, ob),
        function(j1, j2, obs_color) {
          (j1[[obs_color]] + j2[[obs_color]]) / 20
        }
      ),
      jar_ob = pmap_dbl(
        list(chosen_jar, jar1, jar2, ob),
        function(cj, j1, j2, obs_color) {
          count1 <- j1[[obs_color]]
          count2 <- j2[[obs_color]]
          
          probs <- c(count1, count2)
          
          if (cj == 1) {
            return(probs[1])  # P(jar1 | ob)
          } else if (cj == 2) {
            return(probs[2])  # P(jar2 | ob)
          } else {
            return(NA_real_)
          }
        }
      )
    )%>%
    # Now work rowwise: compute prior and likelihood scores
    rowwise() %>%
    mutate(
      # Agent1 (knows 'blue') likelihood scores
      s1_type1 = if (ob == k1) jar1[[ob]] / (jar1[[ob]] + jar2[[ob]])
      else ((total_balls - jar1[k1]) / 2) /
        (((total_balls - jar1[k1]) / 2) + ((total_balls - jar2[k1]) / 2)),
      s1_type2 = if (ob == k1) jar2[[ob]] / (jar1[[ob]] + jar2[[ob]])
      else ((total_balls - jar2[k1]) / 2) /
        (((total_balls - jar1[k1]) / 2) + ((total_balls - jar2[k1]) / 2)),
      
      # Agent2 (knows 'yellow') likelihood scores
      s2_type1 = if (ob == k2) jar1[[ob]] / (jar1[[ob]] + jar2[[ob]])
      else ((total_balls - jar1[k2]) / 2) /
        (((total_balls - jar1[k2]) / 2) + ((total_balls - jar2[k2]) / 2)),
      s2_type2 = if (ob == k2) jar2[[ob]] / (jar1[[ob]] + jar2[[ob]])
      else ((total_balls - jar2[k2]) / 2) /
        (((total_balls - jar1[k2]) / 2) + ((total_balls - jar2[k2]) / 2)),
      
      # Softmax to get likelihoods, then pick based on s1/s2 choice
      #here use a very high tau to simulate a deterministic function (a hard maximum)
      ll_s1 = {
        p <- tau_softmax(c(s1_type1, s1_type2), tau=tau2)
        if (s1 == "Type1") p[1] else p[2]
      },
      ll_s2 = {
        p <- tau_softmax(c(s2_type1, s2_type2), tau=tau2)
        if (s2 == "Type1") p[1] else p[2]
      },
      post = prior_ob * jar_ob * ll_s1 * ll_s2
    ) %>%
    ungroup()
  
  final<-hypotheses %>%
    group_by(chosen_jar)%>%
    summarise(
      avg = sum(post)
    )  %>%
    mutate(posterior = avg / sum(avg))
  
  final <- final %>% filter(chosen_jar == 2)
  
  return(final$posterior)
}

# Function to create all_choice data for a given dataset
create_all_choice <- function(data) {
  conditions <- paste0("p", 1:8)
  all_data <- lapply(1:8, function(i) {
    data %>%
      group_by_at(vars(conditions[i])) %>%
      summarise(count = n()) %>%
      rename(d = conditions[i]) %>%
      mutate(condition = i)
  })
  Reduce(function(x, y) full_join(x, y, by = c("d", "condition", "count")), all_data) %>%
    mutate(d = as.numeric(d))
}

  w_list2 <- c((1+2*n)/(2+2*n), (1+2*n*e)/(2+2*n*e), (1)/(2+2*n), 0.5, (1+2*n)/(2+2*n), (1+2*n*e)/(2+2*n*e), 1/(2+2*n), 0.5)
  w1<-w_list1[i]
  w2<-w_list2[i]
  alpha= 1+w2*ce
  beta= 1+w1*ce
  ll1<-pbeta(1/6,alpha,beta)
  ll2<-pbeta(2/6,alpha,beta)-pbeta(1/6,alpha,beta)
  ll3<-pbeta(3/6,alpha,beta)-pbeta(2/6,alpha,beta)
  ll4<-pbeta(4/6,alpha,beta)-pbeta(3/6,alpha,beta)
  ll5<-pbeta(5/6,alpha,beta)-pbeta(4/6,alpha,beta)
  ll6<-pbeta(6/6,alpha,beta)-pbeta(5/6,alpha,beta)
  #create a list
  ll_values <- c(ll1,ll2,ll3,ll4,ll5,ll6)
  return(ll_values)
}

# Modified objective function to take all_choice as input
objective_function <- function(par, all_choice) {
  ce <- exp(par[1])
  n <- exp(par[2])
  e <- exp(par[3])
  ll_log_sum <- 0
  
  for (m in 1:nrow(all_choice)) {
    likert <- all_choice$d[m]
    condition <- all_choice$condition[m]
    count <- all_choice$count[m]
    sf_lllist <- integral_function3(ce,n,e,condition)
    ll_here <- sf_lllist[likert]
    ll_log_sum <- ll_log_sum - count * log(ll_here)
  }
  return(ll_log_sum)
}

evaluate_round <- function(col_prefix) {
  df <- final_data[, paste0("ll_", col_prefix, c("_nb", "_mv", "_dh", "_none"))]
  colnames(df) <- c("nb", "mv", 'dh', "none")  # 简化列名方便看

  # 计算每行最小值模型
  min_col <- apply(df, 1, function(row) names(which.min(row)))
  min_value <- apply(df, 1, min)

  cat("=== Winner count (", col_prefix, ") ===\n", sep = "")
  print(table(min_col))

  # 赢的模型的负对数似然总和
  min_df <- data.frame(min_col = min_col, min_value = min_value)
  result_sum <- aggregate(min_value ~ min_col, data = min_df, sum)
  cat("=== Loglikelihood sums (", col_prefix, ") ===\n", sep = "")
  print(result_sum)

  # 全体 McFadden R²（每个模型 vs. null）
  ll_sums <- colSums(df)
  logL <- -ll_sums
  logL_null <- logL["none"]
  mcfadden_r2 <- 1 - logL / logL_null
  cat("=== McFadden R² (all data, ", col_prefix, ") ===\n", sep = "")
  print(round(mcfadden_r2, 4))

  # 新增：局部 McFadden R²（每个模型赢的参与者）
  cat("=== Local McFadden R² (only subjects where model won) ===\n")
  unique_models <- unique(min_col)
  for (model in unique_models) {
    idx <- which(min_col == model)
    if (length(idx) > 1) {
      logL_model <- -sum(df[idx, model])
      logL_none  <- -sum(df[idx, "none"])
      r2_local <- 1 - logL_model / logL_none
      cat(model, ": ", round(r2_local, 4), " (", length(idx), " subjects)\n", sep = "")
    } else {
      cat(model, ": <only 1 subject> -- skipping R²\n")
    }
  }
}
