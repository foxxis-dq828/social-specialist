different_agent <- function(k1, k2, s1, s2, tau) {
  jar1 <- list("blue" = k1[1], "yellow" = k1[2], "red" = k1[3])
  jar2 <- list("blue" = k2[1], "yellow" = k2[2], "red" = k2[3])
  jars_list <- list(jar1, jar2)
  colors <- c("blue", "yellow", "red")
  jar_ids <- c(1, 2)
  
  h_list <- list()
  index <- 1
  for (color in colors) {
    for (jar in jar_ids) {
      h_list[[paste0("h_", index)]] <- list("color" = color, "jar" = jar)
      index <- index + 1
    }
  }
  
  softmax <- function(x, tau) {
    exps <- exp(log(x) * tau)
    exps / sum(exps)
  }
  
  jar1_sum <- 0
  jar2_sum <- 0
  
  for (i in seq_along(h_list)) {
    h <- h_list[[i]]
    color <- h[["color"]]
    jar_choice <- h[["jar"]]
    
    # P(s1 | ob), P(s2 | ob)
    if (color == "blue") {
      P_s1_ob <- jars_list[[s1]][["blue"]] / (jars_list[[1]][["blue"]] + jars_list[[2]][["blue"]])
      P_s2_ob <- (sum(unlist(jar1)) - jars_list[[s2]][["yellow"]]) /
        (2 * sum(unlist(jar1)) - jars_list[[1]][["yellow"]] - jars_list[[2]][["yellow"]])
    } else if (color == "yellow") {
      P_s1_ob <- (sum(unlist(jar1)) - jars_list[[s1]][["blue"]]) /
        (2 * sum(unlist(jar1)) - jars_list[[1]][["blue"]] - jars_list[[2]][["blue"]])
      P_s2_ob <- jars_list[[s2]][["yellow"]] / (jars_list[[1]][["yellow"]] + jars_list[[2]][["yellow"]])
    } else {
      P_s1_ob <- (sum(unlist(jar1)) - jars_list[[s1]][["blue"]]) /
        (2 * sum(unlist(jar1)) - jars_list[[1]][["blue"]] - jars_list[[2]][["blue"]])
      P_s2_ob <- (sum(unlist(jar1)) - jars_list[[s2]][["yellow"]]) /
        (2 * sum(unlist(jar1)) - jars_list[[1]][["yellow"]] - jars_list[[2]][["yellow"]])
    }
    
    P_s1_sf <- softmax(c(P_s1_ob, 1 - P_s1_ob), tau)[1]
    P_s2_sf <- softmax(c(P_s2_ob, 1 - P_s2_ob), tau)[1]
    
    h[["P(s1|ob)"]] <- P_s1_sf
    h[["P(s2|ob)"]] <- P_s2_sf
    
    # P(ob)
    total_color <- jars_list[[1]][[color]] + jars_list[[2]][[color]]
    P_ob <- total_color / 40  # 20 balls per jar
    
    # P(jar | ob) using hardmax
    P_jar1_ob <- jars_list[[1]][[color]] / total_color
    P_jar2_ob <- jars_list[[2]][[color]] / total_color
    jar_probs <- c(P_jar1_ob , P_jar2_ob)
    P_jar_given_ob <- jar_probs[jar_choice]
    
    # Final joint probability
    P_joint <- P_ob * P_jar_given_ob * P_s1_sf * P_s2_sf
    h[["P(ob|s1,s2)"]] <- P_joint
    
    # Sum contributions
    if (jar_choice == 1) {
      jar1_sum <- jar1_sum + P_joint
    } else {
      jar2_sum <- jar2_sum + P_joint
    }
    
    h_list[[i]] <- h
  }
  
  jar2_r <- jar2_sum / (jar1_sum + jar2_sum)
  return(jar2_r)
}


same_agent <- function(k1, k2, s1, s2, tau) {
  jar1 <- list("blue" = k1[1], "yellow" = k1[2], "red" = k1[3])
  jar2 <- list("blue" = k2[1], "yellow" = k2[2], "red" = k2[3])
  jars_list <- list(jar1, jar2)
  
  colors <- c("blue", "yellow", "red")
  jar_ids <- c(1, 2)
  
  h_list <- list()
  index <- 1
  for (color in colors) {
    for (jar in jar_ids) {
      h_list[[paste0("h_", index)]] <- list("color" = color, "jar" = jar)
      index <- index + 1
    }
  }
  
  softmax <- function(x, tau) {
    exps <- exp(log(x) * tau)
    exps / sum(exps)
  }
  
  jar1_sum <- 0
  jar2_sum <- 0
  
  for (i in seq_along(h_list)) {
    h <- h_list[[i]]
    color <- h[["color"]]
    jar_choice <- h[["jar"]]
    
    # P(s1 | ob)
    if (color == "blue") {
      P_s1_ob <- jars_list[[s1]][["blue"]] / (jars_list[[1]][["blue"]] + jars_list[[2]][["blue"]])
    } else {
      P_s1_ob <- (sum(unlist(jar1)) - jars_list[[s1]][["blue"]]) /
        (2 * sum(unlist(jar1)) - jars_list[[1]][["blue"]] - jars_list[[2]][["blue"]])
    }
    
    P_s1_sf <- softmax(c(P_s1_ob, 1 - P_s1_ob), tau)[1]
    P_s2_sf <- P_s1_sf  # same agent: s1 == s2
    
    h[["P(s1|ob)"]] <- P_s1_sf
    h[["P(s2|ob)"]] <- P_s2_sf
    
    # Total counts of this color across both jars
    total_color <- jars_list[[1]][[color]] + jars_list[[2]][[color]]
    P_ob <- total_color / 40  # assuming 20 balls per jar
    
    # P(jar | ob) using extreme softmax
    P_jar1_ob <- jars_list[[1]][[color]] / total_color
    P_jar2_ob <- jars_list[[2]][[color]] / total_color
    jar_probs <- c(P_jar1_ob, P_jar2_ob)
    P_jar_given_ob <- jar_probs[jar_choice]
    
    # Final joint probability
    P_joint <- P_ob * P_jar_given_ob * P_s1_sf * P_s2_sf
    h[["P(ob|s1,s2)"]] <- P_joint
    
    if (jar_choice == 1) {
      jar1_sum <- jar1_sum + P_joint
    } else {
      jar2_sum <- jar2_sum + P_joint
    }
    
    h_list[[i]] <- h
  }
  
  jar2_r <- jar2_sum / (jar1_sum + jar2_sum)
  return(jar2_r)
}

w_list2 <- function(tau){
  w_list <- c()
  urn1_list<-list(c(4,15,1),c(1,4,15),c(1,4,15),c(1,4,15))
  urn2_list<-list(c(15,1,4),c(4,15,1),c(4,1,15),c(4,15,1))
  res_div<-list(c(1,2),c(1,1),c(2,2),c(2,1))
  res_nodiv<-list(c(1,1),c(1,1),c(2,2),c(2,2))
  for (i in 1:4){
    urn1<-urn1_list[[i]]
    urn2<-urn2_list[[i]]
    res1_div<-res_div[[i]][[1]]
    res2_div<-res_div[[i]][[2]]
    res1_nodiv<-res_nodiv[[i]][[1]]
    res2_nodiv<-res_nodiv[[i]][[2]]
    w_list[2 * i - 1] <- different_agent(urn1, urn2, res1_div, res2_div, tau)
    w_list[2 * i]    <- same_agent(urn1, urn2, res1_nodiv, res2_nodiv, tau)
  }
  return(w_list)
}

integral_function1 <- function(ce,i,w_list) {
  w <- w_list[i]
  alpha <- 1 + w * ce
  beta <- 1 + (1 - w) * ce
  ll1 <- pbeta(1/6, alpha, beta)
  ll2 <- pbeta(2/6, alpha, beta) - pbeta(1/6, alpha, beta)
  ll3 <- pbeta(3/6, alpha, beta) - pbeta(2/6, alpha, beta)
  ll4 <- pbeta(4/6, alpha, beta) - pbeta(3/6, alpha, beta)
  ll5 <- pbeta(5/6, alpha, beta) - pbeta(4/6, alpha, beta)
  ll6 <- pbeta(6/6, alpha, beta) - pbeta(5/6, alpha, beta)
  ll_values <- c(ll1, ll2, ll3, ll4, ll5, ll6)
  return(ll_values)
}

# Function to create all_choice data for a given dataset
create_all_choice <- function(data) {
  conditions <- paste0("d", 1:8)
  all_data <- lapply(1:8, function(i) {
    data %>%
      group_by_at(vars(conditions[i])) %>%
      summarise(count = n()) %>%
      rename(d = conditions[i]) %>%
      mutate(condition = i)
  })
  Reduce(function(x, y) full_join(x, y, by = c("d", "condition", "count")), all_data) %>%
    mutate(d = as.numeric(d))
}

# Define the Softmax Function
sf <- function(ll_values, tau) {
  # Prevent overflow by subtracting the max before exponentiation
  ll_values <- ll_values - max(ll_values)
  exp_neg_tau_ll <- exp(ll_values * tau)
  probabilities <- exp_neg_tau_ll / sum(exp_neg_tau_ll)
  return(probabilities)
}

evaluate_round <- function(col_prefix) {
  df <- final_data[, paste0("ll_", col_prefix, c("_nb", "_mv", "_dh", "_none"))]
  colnames(df) <- c("nb", "mv", "dh", "none")  # 简化列名方便看

  # 计算每行最小值模型
  min_col <- apply(df, 1, function(row) names(which.min(row)))
  min_value <- apply(df, 1, min)

  cat("=== Winner count (", col_prefix, ") ===\n", sep = "")
  print(table(min_col))

  # 赢的模型的负对数似然总和
  min_df <- data.frame(min_col = min_col, min_value = min_value)
  result_sum <- aggregate(min_value ~ min_col, data = min_df, sum)
  cat("=== Loglikelihood sums (", col_prefix, ") ===\n", sep = "")
  print(result_sum)

  # 全体 McFadden R²（每个模型 vs. null）
  ll_sums <- colSums(df)
  logL <- -ll_sums
  logL_null <- logL["none"]
  mcfadden_r2 <- 1 - logL / logL_null
  cat("=== McFadden R² (all data, ", col_prefix, ") ===\n", sep = "")
  print(round(mcfadden_r2, 4))

  # 新增：局部 McFadden R²（每个模型赢的参与者）
  cat("=== Local McFadden R² (only subjects where model won) ===\n")
  unique_models <- unique(min_col)
  for (model in unique_models) {
    idx <- which(min_col == model)
    if (length(idx) > 1) {
      logL_model <- -sum(df[idx, model])
      logL_none  <- -sum(df[idx, "none"])
      r2_local <- 1 - logL_model / logL_none
      cat(model, ": ", round(r2_local, 4), " (", length(idx), " subjects)\n", sep = "")
    } else {
      cat(model, ": <only 1 subject> -- skipping R²\n")
    }
  }
}

    arrange(d)
